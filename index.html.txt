        async function callAI(prompt) {
            const key = document.getElementById('apiKey').value.trim() || localStorage.getItem(API_KEY_STORAGE);
            const provider = document.getElementById('providerSelect').value;
            const model = document.getElementById('modelSelect').value;

            if (!key) {
                alert('⚠️ Clé API manquante');
                throw new Error('No API key');
            }

            const response = await fetch(provider === 'anthropic' ? ANTHROPIC_API_URL : `${GEMINI_BASE_URL}/${model}:generateContent?key=${key}`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    ...(provider === 'anthropic' && { 'x-api-key': key, 'anthropic-version': '2023-06-01', 'anthropic-dangerous-direct-browser-access': 'true' })
                },
                body: JSON.stringify(provider === 'anthropic' ?
                    { model, max_tokens: 4000, messages: [{ role: 'user', content: prompt }] } :
                    { contents: [{ parts: [{ text: prompt }] }], generationConfig: { temperature: 0.4, maxOutputTokens: 4000 } }
                )
            });

            if (!response.ok) throw new Error(`Erreur ${response.status}`);

            const data = await response.json();
            return provider === 'anthropic' ?
                data.content.find(c => c.type === 'text')?.text || '' :
                data.candidates?.[0]?.content?.parts?.[0]?.text || '';
        }
